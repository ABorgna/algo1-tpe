\aux{min}{xs : [\ent]}{\ent}{
    cab([n \:|\: n \selec xs, (\forall x \selec xs) n \le xs])
}

\aux{max}{xs : [\ent]}{\ent}{
    cab([n \:|\: n \selec xs, (\forall x \selec xs) n \ge xs])
}

\aux{rango}{ns : [\ent]}{[\ent]}{
    [min(xs) \ldots max(xs)]
}

\aux{find}{xs : [T], x : T}{\ent}{
    \sum [1 \: | \: i \selec [0 \ldots |xs|), x \not\in xs[\ldots i] \:]
}
